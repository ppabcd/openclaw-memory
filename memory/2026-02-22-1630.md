```
#!/usr/bin/env python3
import boto3
from datetime import datetime

CREDS_FILE = "/root/.openclaw/workspace/.r2_credentials"
BATCH_SIZE = 1000

def load_credentials():
    creds = {}
    with open(CREDS_FILE) as f:
        for line in f:
            line = line.strip()
            if "=" in line:
                k, v = line.split("=", 1)
                creds[k] = v
    return creds

def get_client():
    creds = load_credentials()
    s3 = boto3.client(
        "s3",
        endpoint_url=creds["R2_ENDPOINT"],
        aws_access_key_id=creds["R2_ACCESS_KEY_ID"],
        aws_secret_access_key=creds["R2_SECRET_ACCESS_KEY"],
        region_name="auto",
    )
    return s3, creds["R2_BUCKET"]

def delete_batch():
    client, bucket = get_client()
    total_deleted = 0
    total_size = 0
    batch_no = 0

    while True:
        resp = client.list_objects_v2(Bucket=bucket, MaxKeys=BATCH_SIZE)
        contents = resp.get("Contents", [])
        if not contents:
            print("\nBucket empty!")
            break

        keys = [{"Key": obj["Key"]} for obj in contents]
        batch_size_mb = sum(obj["Size"] for obj in contents) / (1024**2)
        batch_no += 1

        result = client.delete_objects(Bucket=bucket, Delete={"Objects": keys})
        deleted = len(result.get("Deleted", []))
        total_deleted += deleted
        total_size += batch_size_mb

        print(f"Batch {batch_no:>4}: deleted {deleted:>4} files | {batch_size_mb:>6.2f} MB "
              f"| total {total_deleted:,} files ({total_size:>6.2f} MB)")

        if len(contents) < BATCH_SIZE:
            print("\nReached end of bucket.")
            break

if __name__ == "__main__":
    print(f"Started at {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
    delete_batch()
```
